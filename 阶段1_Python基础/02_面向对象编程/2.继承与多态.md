# 继承与多态

## 一、继承：代码复用的核心机制

继承让子类**自动拥有**父类的属性和方法，同时可以**添加新功能**或**修改已有行为**。

```
父类（基类）
    └── 子类（派生类）：继承父类的一切 + 可以扩展/修改
```

### 基本语法

```python
class Animal:
    """父类"""
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def eat(self):
        print(f"{self.name} 在吃东西")

    def sleep(self):
        print(f"{self.name} 在睡觉")

    def __str__(self):
        return f"{self.__class__.__name__}({self.name}, {self.age}岁)"


class Dog(Animal):    # Dog 继承 Animal
    """子类"""
    def bark(self):   # 新增方法
        print(f"{self.name} 说：汪汪！")


class Cat(Animal):    # Cat 也继承 Animal
    def meow(self):
        print(f"{self.name} 说：喵喵！")


# 使用
dog = Dog("旺财", 3)
dog.eat()     # 继承自 Animal：旺财 在吃东西
dog.sleep()   # 继承自 Animal：旺财 在睡觉
dog.bark()    # Dog 自己的方法：旺财 说：汪汪！
print(dog)    # Dog(旺财, 3岁)
```

## 二、super() —— 调用父类方法

当子类需要**扩展**父类方法（而不是完全替换）时，用 `super()` 调用父类版本。

```python
class Animal:
    def __init__(self, name, age):
        self.name = name
        self.age = age

class Dog(Animal):
    def __init__(self, name, age, breed):
        super().__init__(name, age)  # 先调用父类 __init__，初始化 name 和 age
        self.breed = breed           # 再初始化子类独有的属性

    def info(self):
        return f"{self.name}（{self.breed}），{self.age}岁"

dog = Dog("旺财", 3, "金毛")
print(dog.info())   # 旺财（金毛），3岁
```

**不用 super() 的坑**：

```python
class Dog(Animal):
    def __init__(self, name, age, breed):
        # 忘记调用 super().__init__()
        self.breed = breed
        # 结果：self.name 和 self.age 没被初始化 → AttributeError！
```

## 三、方法重写（Override）

子类可以**覆盖**父类的方法，提供不同的实现。

```python
class Animal:
    def speak(self):
        print("动物发出声音")

class Dog(Animal):
    def speak(self):          # 重写父类的 speak
        print("汪汪！")

class Cat(Animal):
    def speak(self):          # 重写父类的 speak
        print("喵喵！")

class Duck(Animal):
    def speak(self):
        print("嘎嘎！")


# 多态：同一个方法调用，不同对象有不同行为
animals = [Dog(), Cat(), Duck()]
for animal in animals:
    animal.speak()
# 汪汪！
# 喵喵！
# 嘎嘎！
```

## 四、多态：同一接口，不同实现

多态的本质：**不关心对象是什么类型，只关心它能做什么**。

```python
class Shape:
    def area(self) -> float:
        raise NotImplementedError("子类必须实现 area()")

    def perimeter(self) -> float:
        raise NotImplementedError("子类必须实现 perimeter()")

    def describe(self):
        """共用方法，依赖子类实现的 area()"""
        print(f"{self.__class__.__name__}: 面积={self.area():.2f}, 周长={self.perimeter():.2f}")


class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        return 3.14159 * self.radius ** 2

    def perimeter(self):
        return 2 * 3.14159 * self.radius


class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height

    def area(self):
        return self.width * self.height

    def perimeter(self):
        return 2 * (self.width + self.height)


class Triangle(Shape):
    def __init__(self, a, b, c):
        self.a, self.b, self.c = a, b, c

    def area(self):
        s = (self.a + self.b + self.c) / 2
        return (s * (s-self.a) * (s-self.b) * (s-self.c)) ** 0.5

    def perimeter(self):
        return self.a + self.b + self.c


# 多态：统一接口处理不同类型
shapes = [Circle(5), Rectangle(4, 6), Triangle(3, 4, 5)]
for shape in shapes:
    shape.describe()   # 每个 shape 调用自己的 area() 和 perimeter()

# 计算总面积（不关心具体类型）
total = sum(s.area() for s in shapes)
print(f"总面积: {total:.2f}")
```

## 五、封装：控制访问权限

Python 通过**命名约定**实现访问控制（没有真正的 private）：

```python
class BankAccount:
    def __init__(self, owner, balance):
        self.owner = owner          # 公开属性（public）
        self._balance = balance     # 受保护属性（protected）：约定不直接访问
        self.__pin = "1234"         # 私有属性（private）：名称改写，外部难以直接访问

    def deposit(self, amount):
        if amount > 0:
            self._balance += amount
            print(f"存入 {amount} 元，余额: {self._balance}")

    def withdraw(self, amount, pin):
        if pin != self.__pin:       # 内部可以访问 __pin
            raise ValueError("PIN 错误！")
        if amount > self._balance:
            raise ValueError("余额不足！")
        self._balance -= amount
        print(f"取出 {amount} 元，余额: {self._balance}")

    def get_balance(self):
        return self._balance        # 通过方法访问，可以加权限控制


account = BankAccount("张三", 1000)
account.deposit(500)

# 外部访问 _balance（约定上不应该这样做，但 Python 不强制）
print(account._balance)   # 还是能访问，只是约定

# 外部访问 __pin（名称被改写，不能直接访问）
# print(account.__pin)      # ← AttributeError
print(account._BankAccount__pin)   # 还是能绕过，但很麻烦 → 所以说"无真正私有"
```

| 命名 | 含义 | 外部可访问？ |
|------|------|------------|
| `name` | 公开 | ✅ 随意访问 |
| `_name` | 约定受保护 | ✅ 技术上可以，约定不要 |
| `__name` | 名称改写 | 需要用 `_类名__name` 才能访问 |

## 六、多重继承

Python 支持一个类继承多个父类（很多语言不支持）：

```python
class Flyable:
    def fly(self):
        print(f"{self.name} 在飞！")

class Swimmable:
    def swim(self):
        print(f"{self.name} 在游泳！")

class Duck(Animal, Flyable, Swimmable):
    """鸭子：既能飞又能游泳"""
    def __init__(self, name, age):
        super().__init__(name, age)
    
    def speak(self):
        print(f"{self.name} 说：嘎嘎！")

duck = Duck("唐老鸭", 2)
duck.speak()   # 嘎嘎！
duck.fly()     # 在飞！
duck.swim()    # 在游泳！
```

**MRO（方法解析顺序）**：多重继承时，Python 按 C3 线性化算法确定方法查找顺序：

```python
print(Duck.__mro__)
# (<class 'Duck'>, <class 'Animal'>, <class 'Flyable'>, <class 'Swimmable'>, <class 'object'>)
```

## 七、后端开发中的继承实例

```python
# FastAPI 中的异常继承体系
class AppException(Exception):
    """应用基础异常"""
    def __init__(self, message: str, code: int = 400):
        self.message = message
        self.code = code

class AuthException(AppException):
    """认证异常"""
    def __init__(self, message: str = "认证失败"):
        super().__init__(message, code=401)

class PermissionException(AppException):
    """权限异常"""
    def __init__(self, message: str = "权限不足"):
        super().__init__(message, code=403)

class NotFoundException(AppException):
    """资源不存在"""
    def __init__(self, resource: str):
        super().__init__(f"{resource} 不存在", code=404)

# 使用
try:
    raise NotFoundException("用户")
except AppException as e:   # 统一捕获所有应用异常
    print(f"错误 {e.code}: {e.message}")
```
